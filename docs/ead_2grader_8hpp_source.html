42  NodeptrT<T> dlocal (ead::NodeptrT<T> orig, size_t idx)
43  {
44  auto f = static_cast<ade::iFunctor*>(orig->get_tensor().get());
45  const ade::ArgsT& args = f->get_children();
46  NodeptrT<T> out = nullptr;
47  switch ((age::_GENERATED_OPCODE) f->get_opcode().code_)
48  {
49  case age::ABS:
50  out = age::div(ead::to_node<T>(args[0].get_tensor()), orig);
51  break;
52  case age::NEG:
53  out = ead::make_constant_scalar<T>(-1, args[0].get_tensor()->shape());
54  break;
55  case age::SIN:
56  out = age::cos(ead::to_node<T>(args[0].get_tensor()));
57  break;
58  case age::COS:
59  out = age::neg(age::sin(ead::to_node<T>(args[0].get_tensor())));
60  break;
61  case age::TAN:
62  out = age::div(
63  ead::make_constant_scalar<T>(1,args[0].get_tensor()->shape()),
64  age::pow(
65  age::cos(ead::to_node<T>(args[0].get_tensor())),
66  ead::make_constant_scalar<T>(2, args[0].get_tensor()->shape())
67  )
68  );
69  break;
70  case age::EXP:
71  out = orig;
72  break;
73  case age::LOG:
74  out = age::div(
75  ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape()),
76  ead::to_node<T>(args[0].get_tensor())
77  );
78  break;
79  case age::SQRT:
80  out = age::div(
81  ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape()),
82  age::mul(
83  ead::make_constant_scalar<T>(2,args[0].get_tensor()->shape()),
84  orig
85  )
86  );
87  break;
88  case age::SQUARE:
89  out = age::mul(
90  ead::make_constant_scalar<T>(2,args[0].get_tensor()->shape()),
91  ead::to_node<T>(args[0].get_tensor())
92  );
93  break;
94  case age::CUBE:
95  out = age::mul(
96  ead::make_constant_scalar<T>(3, args[0].get_tensor()->shape()),
97  age::square(ead::to_node<T>(args[0].get_tensor()))
98  );
99  break;
100  case age::ROUND:
101  out = ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape());
102  break;
103  case age::SIGMOID:
104  out = age::sigmoid_grad(ead::to_node<T>(args[0].get_tensor()));
105  break;
106  case age::SIGMOID_GRAD:
107  out = age::mul(
108  orig,
109  age::sub(
110  ead::make_constant_scalar<T>(1,args[0].get_tensor()->shape()),
111  age::mul(
112  ead::make_constant_scalar<T>(2, args[0].get_tensor()->shape()),
113  age::sigmoid(ead::to_node<T>(args[0].get_tensor()))
114  )
115  )
116  );
117  break;
118  case age::TANH:
119  out = age::sub(
120  ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape()),
121  age::square(orig)
122  );
123  break;
124  case age::ADD:
125  out = ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape());
126  break;
127  case age::MUL:
128  out = ead::to_node<T>(args[(size_t)(idx==0)].get_tensor());
129  break;
130  case age::MAX:
131  case age::MIN:
132  out = age::eq(orig, ead::to_node<T>(args[idx].get_tensor()));
133  break;
134  case age::POW:
135  out = idx==0 ?
136  age::mul(
137  ead::to_node<T>(args[1].get_tensor()),
138  age::pow(
139  ead::to_node<T>(args[0].get_tensor()),
140  age::sub(
141  ead::to_node<T>(args[1].get_tensor()),
142  ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape())
143  )
144  )
145  ) :
146  age::mul(age::log(ead::to_node<T>(args[0].get_tensor())), orig);
147  break;
148  case age::SUB:
149  out = ead::make_constant_scalar<T>(idx == 0 ? 1 : -1, args[0].get_tensor()->shape());
150  break;
151  case age::DIV:
152  {
153  auto denom = ead::to_node<T>(args[1].get_tensor());
154  out = idx==0 ?
155  age::div(
156  ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape()),
157  denom
158  ) :
159  age::div(
160  age::div(age::neg(ead::to_node<T>(args[0].get_tensor())), denom),
161  denom
162  );
163  }
164  break;
165  case age::EQ:
166  case age::NEQ:
167  case age::GT:
168  case age::LT:
169  case age::RAND_UNIF:
170  out = ead::make_constant_scalar<T>(0, args[0].get_tensor()->shape());
171  break;
172  case age::REDUCE_SUM:
173  case age::EXTEND:
174  case age::PERMUTE:
175  out = ead::make_constant_scalar<T>(1, args[0].get_tensor()->shape());
176  break;
177  case age::REDUCE_PROD:
178  out = age::div(
179  reduce_grad(args[0], orig, idx),
180  to_node<T>(args[0].get_tensor())
181  );
182  break;
183  case age::REDUCE_MAX:
184  case age::REDUCE_MIN:
185  out = age::eq(
186  reduce_grad(args[0], orig, idx),
187  to_node<T>(args[0].get_tensor())
188  );
189  break;
190  case age::MATMUL:
191  {
192  NodeptrT<T> lhs = to_node<T>(args[0].get_tensor());
193  NodeptrT<T> rhs = to_node<T>(args[1].get_tensor());
194  out = 0 == idx ?
195  // ext_rhs
196  age::permute(age::extend(rhs, 2, {lhs->shape().at(1)}), {0,2,1}) :
197  // ext_lhs
198  age::permute(age::extend(lhs, 2, {rhs->shape().at(0)}), {2,1,0});
199  }
200  break;
201  case age::CONV:
202  default:
203  logs::fatal("Unknown op");
204  }
205  return out;
206  }

208  NodeptrT<T> chain (NodeptrT<T> local, ade::iFunctor* fwd, NodeptrT<T> bwd, size_t idx)
209  {
210  NodeptrT<T> out;
211  switch (fwd->get_opcode().code_)
212  {
213  case age::ABS:
214  case age::NEG:
215  case age::SIN:
216  case age::COS:
217  case age::TAN:
218  case age::EXP:
219  case age::LOG:
220  case age::SQRT:
221  case age::SQUARE:
222  case age::CUBE:
223  case age::ROUND:
224  case age::SIGMOID:
225  case age::SIGMOID_GRAD:
226  case age::TANH:
227  case age::ADD:
228  case age::MUL:
229  case age::MAX:
230  case age::MIN:
231  case age::POW:
232  case age::SUB:
233  case age::DIV:
234  case age::EQ:
235  case age::NEQ:
236  case age::GT:
237  case age::LT:
238  case age::RAND_UNIF:
239  out = age::mul(local, bwd);
240  break;
241  case age::REDUCE_MAX:
242  case age::REDUCE_MIN:
243  case age::REDUCE_PROD:
244  case age::REDUCE_SUM:
245  out = age::mul(local, reduce_grad(fwd->get_children()[0], bwd, idx));
246  break;
247  case age::EXTEND:
248  out = age::mul(local, extend_grad(fwd, bwd, idx));
249  break;
250  case age::PERMUTE:
251  out = age::mul(local, permute_grad(fwd, bwd, idx));
252  break;
253  case age::MATMUL:
254  out = age::reduce_sum(
255  age::permute(
256  age::mul(local,
257  age::extend(bwd, 2, {
258  fwd->get_children()[0].get_tensor()->shape().at(0)
259  })),
260  0 == idx ?
261  std::vector<uint8_t>{2, 1, 0} :
262  std::vector<uint8_t>{0, 2, 1}), 2);
263  break;
264  case age::CONV:
265  default:
266  logs::fatal("Unknown op");
267  }
268  return out;
269  }

273 template <typename T>
274 NodeptrT<T> derive_with_edges (ade::EdgesT& edges, NodeptrT<T> root, NodeptrT<T> target)
275 {
276  if (root->get_tensor() == target->get_tensor())
277  {
278  return make_constant_scalar((T) 1, target->shape());
279  }
280 
281  ade::iTensor* target_tens = target->get_tensor().get();
282  ade::PathFinder finder(target_tens);
283  root->get_tensor()->accept(finder);
284 
285  auto& pathmap = finder.parents_;
286  // no path to wrt
287  if (pathmap.empty())
288  {
289  return make_constant_scalar((T) 0, target->shape());
290  }
291  // else there exists a path to wrt
292  // using pathfinder, breadth first traverse from this to wrt
293  std::string source_str = root->get_tensor()->to_string();
294  ade::GraphStat stat;
295  root->get_tensor()->accept(stat);
296  auto owners = ade::track_owners(root->get_tensor());
297 
298  std::list<ade::iFunctor*> parents;
299  std::transform(pathmap.begin(), pathmap.end(),
300  std::back_inserter(parents),
301  [](std::pair<ade::iTensor*,std::unordered_set<size_t>> parent)
302  {
303  return static_cast<ade::iFunctor*>(parent.first);
304  });
305  parents.sort(
306  [&](ade::iFunctor* a, ade::iFunctor* b)
307  {
308  return stat.graphsize_[a] > stat.graphsize_[b];
309  });
310 
311  auto root_grad = make_constant_scalar((T) 1, root->shape());
312  std::unordered_map<const ade::iTensor*,NodesT<T>> grads = {
313  {root->get_tensor().get(), {root_grad}}
314  };
315  ChainRuler<T> ruler;
316  for (ade::iFunctor* parent : parents)
317  {
318  NodesT<T>& gargs = grads[parent];
319  NodeptrT<T> bwd = gargs[0];
320  for (size_t i = 1, n = gargs.size(); i < n; ++i)
321  {
322  bwd = age::add(bwd, gargs[i]);
323  }
324  edges.push_back(ade::Edge{
325  bwd->get_tensor(),
326  owners[parent],
327  ade::Opcode{
328  fmts::sprintf("GRADIENT_%s_%s",
329  source_str.c_str(), parent->to_string().c_str()),
330  GRADIENT
331  }
332  });
333 
334  auto& grad_indices = pathmap[parent];
335  ade::ArgsT children = parent->get_children();
336  size_t nchildren = children.size();
337  // assert: all nnary-children use identity mapping,
338  // so no children-arg is direct mapping
339  ade::TensT args(nchildren);
340  std::transform(children.begin(), children.end(), args.begin(),
341  [](ade::FuncArg& arg)
342  {
343  return arg.get_tensor();
344  });
345  // for each painted child, calculate dThis/dChild
346  // go through grads in order
347  std::list<size_t> ordered(grad_indices.begin(), grad_indices.end());
348  ordered.sort();
349  for (size_t i : ordered)
350  {
351  auto local = ruler.dlocal(ead::to_node<T>(owners[parent].lock()), i); // todo: store this for reuse
352  edges.push_back(ade::Edge{
353  local->get_tensor(),
354  owners[parent],
355  ade::Opcode{
356  fmts::sprintf("GRADIENT_%s_%s",
357  parent->to_string().c_str(), args[i]->to_string().c_str()),
358  GRADIENT
359  }
360  });
361  auto grad_step = ruler.chain(local, parent, bwd, i);
362  // auto grad_step = age::chain_rule<T>(parent, bwd, args, i);
363  grads[args[i].get()].push_back(grad_step);
364  }
365  }
366  NodesT<T>& outargs = grads[target_tens];
367  NodeptrT<T> out = outargs[0];
368  for (size_t i = 1, n = outargs.size(); i < n; ++i)
369  {
370  out = age::add(out, outargs[i]);
371  }
372  std::string target_str;
373  if (auto target_v = dynamic_cast<Variable<T>*>(target_tens))
374  {
375  target_str = target_v->label_;
376  }
377  else
378  {
379  target_str = target_tens->to_string();
380  }
381  edges.push_back(ade::Edge{
382  out->get_tensor(),
383  target->get_tensor(),
384  ade::Opcode{
385  fmts::sprintf("GRADIENT_%s_%s",
386  source_str.c_str(), target_str.c_str()),
387  GRADIENT
388  }
389  });
390  return out;
391 }
