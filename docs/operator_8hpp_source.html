155 void permute (T* out, const T* in, ade::Shape outshape, ade::Shape shape,
156  std::vector<uint8_t> order)
157 {
158  size_t n = shape.n_elems();
159  uint8_t orig_rank = shape.n_rank();
160  bool visited[ade::rank_cap];
161  std::memset(visited, false, sizeof(ade::DimT) * ade::rank_cap);
162  for (size_t i = 0, n = order.size(); i < n; ++i)
163  {
164  visited[order[i]] = true;
165  }
166  for (size_t i = 0, n = orig_rank; i < n; ++i)
167  {
168  if (false == visited[i])
169  {
170  order.push_back(i);
171  }
172  }
173 
174  uint8_t norder = order.size();
175  std::vector<ade::DimT> coords(orig_rank);
176  std::vector<ade::DimT> converted(norder);
177  for (ade::NElemT srci = 0; srci < n; ++srci)
178  {
179  coords = coordinate(shape, srci);
180

182  {
183  converted[i] = coords[order[i]];
184  }
185 
186  ade::NElemT desti = index(outshape, converted);
187  out[desti] = in[srci];
