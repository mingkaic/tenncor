145 {
146  size_t nargs = args.size();
147  CommCandsT<T> candidates = {{
148  RuleContext<T>(), std::vector<bool>(nargs, false)
149  }};
150  for (auto& sub_rule : sub_rules)
151  {
152  CommCandsT<T> next_cands;
153  for (auto& cand_pair : candidates)
154  {
155  auto& field = cand_pair.second;
156  for (size_t j = 0; j < nargs; ++j)
157  {
158  if (field[j])
159  {
160  continue;
161  }
162  RuleContext<T> temp_ctx = cand_pair.first;
163  if (args[j].arg_->rulify(temp_ctx, sub_rule.arg_))
164  {
165  std::vector<bool> temp_field = field;
166  temp_field[j] = true;
167  next_cands.push_back({temp_ctx, temp_field});
168  }
169  }
170  }
171  candidates = next_cands;
172  }
