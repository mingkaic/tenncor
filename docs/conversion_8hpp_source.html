150 void apply_rules (Representer<T>& repr, UniqueTensT& unique_tens,
151  const ConversionsT<T>& conversions)
152 {

155  std::list<RepptrT<T>> bottom_up;
156  {
157  size_t rule_minheight = 0;
158  {
159  std::vector<size_t> rule_minheights;
160  rule_minheights.reserve(conversions.size());
161  std::transform(conversions.begin(), conversions.end(),
162  std::back_inserter(rule_minheights),
163  [](const RuleConversion<T>& rule_convs)
164  {
165  return rule_convs.source_->get_minheight();
166  });
167  if (false == rule_minheights.empty())
168  {
169  rule_minheight = *std::min_element(
170  rule_minheights.begin(), rule_minheights.end());
171  }
172  }
173 
174  std::unordered_map<RepptrT<T>,NumRange<size_t>> heights;
175  for (auto reppair : repr.reps_)
176  {
177  auto& tens = reppair.first;
178  auto& rep = reppair.second;
179  NumRange<size_t> repheights = rep->get_heights();
180  heights.emplace(rep, repheights);
181  reverse_rep[rep].push_back(tens);
182  ownermap.emplace(rep.get(), rep);
183  }
184  for (auto reppair : heights)
185  {
186  if (reppair.second.lower_ >= rule_minheight)
187  {
188  bottom_up.push_back(reppair.first);
189  }
190  }
191  bottom_up.sort(
192  [&](RepptrT<T>& a, RepptrT<T>& b)
193  {
194  return heights.at(a).upper_ < heights.at(b).upper_;
195  });
196  }
197 
198  ConversionsT<T> valid_conversions;
199  valid_conversions.reserve(conversions.size());
200  std::copy_if(conversions.begin(), conversions.end(),
201  std::back_inserter(valid_conversions),
202  [](const RuleConversion<T>& conv)
203  {
204  return nullptr != conv.target_ && nullptr != conv.source_;
205  });
206  if (valid_conversions.size() < conversions.size())
207  {
208  logs::error("applying rules with in invalid conversions");
209  }
210 
211  for (RepptrT<T> rep : bottom_up)
212  {
213  auto& tens_keys = reverse_rep[rep];
214  // there must be at least 1 tens mapped to tens in representer
215  assert(tens_keys.size() > 0);
216  ade::Shape exshape = tens_keys[0]->shape();
217  std::string orig_id = rep->get_identifier();
218  for (const RuleConversion<T>& conv : valid_conversions)
219  {
220  RuleContext<T> ctx;
221  if (rep->rulify(ctx, conv.source_))
222  {
223  // match found, convert
224  logs::debugf("applying conversion %s", conv.str_form_.c_str());
225  rep = conv.target_->make_rep(repr, ownermap, ctx, exshape);
226  if (nullptr == rep)
227  {
228  std::vector<T> zeros(exshape.n_elems(), 0);
229  rep = std::make_shared<ConstRep<T>>(zeros.data(), exshape);
230  }
231  }
232  }

234  std::string fresh_id = rep->get_identifier();
235  if (orig_id != fresh_id)
236  {

238  auto it = unique_tens.find(fresh_id);
239  if (unique_tens.end() == it)
240  {
241  // no existing imprint, associate new rep
242  unique_tens.emplace(fresh_id, tens_keys[0]);
243  }
244  else

246  // found existing imprint of rep, use existing rep instead
247  rep = repr.reps_[it->second];
248  }

252  }
